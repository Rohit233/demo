import java.util.Arrays;
import java.util.Scanner;

public class Process {
    static double[] wt ;
    static double[] tat;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("Enter number of processes");
        int n = sc.nextInt();
        wt = new double[n];
        tat = new double[n];
        int[] processTimeBurst = new int[n];
        System.out.println("Enter processes");
        for(int i=0; i<n; i++){
            processTimeBurst[i] = sc.nextInt();
        }

        double avt = averageWaitingTime(processTimeBurst);
        double tt = averageTurnaroundTime(processTimeBurst);

        System.out.println("Waiting time is : ");
        printProcess(wt);
        System.out.println("Turn Around Time is : ");
        printProcess(tat);


        System.out.println("Average Waiting Time is "+avt);

        System.out.println("Avergae TurnAround Time is "+tt);

    }

    public static void printProcess(double arr[]){
        for(int i=0; i<arr.length; i++){
            System.out.println("P"+(i+1)+" = "+arr[i]);
        }
    }

    public static double averageWaitingTime(int[] processTimeBurst){
        double avt = 0;
        double avtTest = 0;
        for(int i = 0;i<processTimeBurst.length-1;i++ ) {
            wt[i] = avt;
            avt = avt + processTimeBurst[i];
            avtTest += avt;


        }
        wt[processTimeBurst.length-1] = avt;
        avt  = avtTest/processTimeBurst.length;

        return avt;
    }

    public  static double averageTurnaroundTime(int[] processTimeBurst){
        double tt = processTimeBurst[0];
        tat[0] = processTimeBurst[0];
        int att = processTimeBurst[0];
        for(int i = 1;i<processTimeBurst.length;i++ ){
            tt += processTimeBurst[i];
            att += tt;
            tat[i] = tt;
//            System.out.println(tt);
        }
        tt  = att/processTimeBurst.length;
        return tt;
    }
}
