# BEST FIT,WORST FIT,FIRST FIT (PR-5)

import java.util.Arrays;
class HelloWorld {
    public static void main(String[] args) {
        int [] block = {2,500,90,200,10,300,200,50,600,500,1000};
        int programSize = 40;
        System.out.println("Fitst Fit: " + getFirstFit(block,programSize));
        System.out.println("Best Fit: " + getBestFit(block,programSize));
        System.out.println("Worst Fit: " + getWorstFit(block,programSize));
    }
    
    static public int getWorstFit(int[] block,int programSize){
        return block[block.length-1];
    }        
    
    static public int getBestFit(int[] block,int programSize){
        Arrays.sort(block);
        for(int i: block){
            if(i >= programSize){
                return i;
            }
        }
        return -1;
    }
    
    static public int getFirstFit(int[] block,int programSize){
         for(int i : block){
             if(i >= programSize){
                 return i;
             }
         }
         return -1;
    }
    
}
